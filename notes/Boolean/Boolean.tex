\section{Laws}

% Associativity
% Commutativity
% Distribuibity

\begin{align*}
    x \lor 0 = x \\
    x \land 0 = 0 \\
    x \lor 1 = 1 \\
    x \land 1 = x \\
    x \lor x = x \\
    x \land x = x \\
    x \land (x \lor y) = x \\
    x \lor (x \land y) = x \\
    x \land \lnot x = 0 \\
    x \lor \lnot x = 1 \\
    \lnot x \land \lnot y = \lnot (x \lor y) \\
    \lnot x \lor \lnot y = \lnot (x \land y)
\end{align*}
