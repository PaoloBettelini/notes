        \item \((\integers / n, +) = \gengrp{{[1]}_n}\);
        \item \(\gengrp{\text{rotations in }\dihedralgrp_{2n}}\)
            is a \cyclicgroup of the roation \(\frac{2\picircle}{n}\);
    \end{enumerate}
\end{snippetexample}
